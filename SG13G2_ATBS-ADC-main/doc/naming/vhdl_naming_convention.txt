NAMING CONVENTION VHDL - ISP

------------------------------------------------------------------------------
-- file naming (suffix)
------------------------------------------------------------------------------
_ea 	  ... entity + architecture
_e		  ... entity
_a		  ... architecture
_tb		  ...	testbench
_p		  ... package
_board	... top level entity + architecture

------------------------------------------------------------------------------
-- entity/architecture naming
------------------------------------------------------------------------------
bhv		... architecture testbench / not synthesizable
rtl		... architecture synthesizable

entity name = file name without suffix

------------------------------------------------------------------------------
-- port signals
------------------------------------------------------------------------------
_i			... input
_o			... output
_n_i		... negated input

clock_i
reset_i

!!comment functionality -- important for std_ulogic flags

------------------------------------------------------------------------------
-- register names (suffix)
------------------------------------------------------------------------------
name_reg

------------------------------------------------------------------------------
-- combinatorics names (suffix)
------------------------------------------------------------------------------
name_logic

------------------------------------------------------------------------------
-- register signals
------------------------------------------------------------------------------
signal <= next_signal (! no curr_signal)

------------------------------------------------------------------------------
-- negated signal (suffix)
------------------------------------------------------------------------------
_n

------------------------------------------------------------------------------
-- enable/ disable (prefix)
------------------------------------------------------------------------------
enable_
disable_

------------------------------------------------------------------------------
-- strobes (suffix)
------------------------------------------------------------------------------
nameofsignal_strb ... name of signal to be validated with strb suffix

------------------------------------------------------------------------------
-- entity implementation labels
------------------------------------------------------------------------------
dut_			      ... prefix (embedded enitity in testbench)
nameofentity_0	... suffix with continuous number (embedded entity in architecture)

------------------------------------------------------------------------------
-- type definitions
------------------------------------------------------------------------------
T				... starts with capital Letter T (e.g. type TShiftReg, TState)

------------------------------------------------------------------------------
-- edge detection (suffix)
------------------------------------------------------------------------------
_edge_f			... falling edge
_edge_r			... rising edge
